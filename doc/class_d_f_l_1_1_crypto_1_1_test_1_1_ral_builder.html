Detailed Description

Generates a RAL string from given attributes.

Details on the string format can be found in DFL::Crypto::RoleAuthorizationList.

const auto privKey{PrivateKey::generateRsaKey()};
const auto ral{
RoleAuthorizationList{Test::RalBuilder{}
.comment("Homer's Role Authorization List")
.roles(Test::KnownRootCas::draegerTest(),
Oids{Oid{Oid::DotNotation{"1.2.3"}}})
.signature(privKey)
.str()}};
const auto _{Test::MockRalPublicKey{PublicKey{privKey}}};
ASSERT_TRUE(ral.valid());
