135 template<class T_ParentEntity, class T_EntityAccess>
136 inline void MdibChildIterator<T_ParentEntity, T_EntityAccess>::setCurrentItem()
137 {
138  m_currentEntity = boost::none;
139  if (m_handleIt == m_handleEnd || !m_entityAccess)
140  {
141  return;
142  }
143 
144  while (m_handleIt != m_handleEnd)
145  {
146  T_ParentEntity::ChildEntities::applyVisitorWithEarlyExit(SetCurrentItemVisitor{}, *m_entityAccess, *m_handleIt, m_currentEntity);
147  if (m_currentEntity)
148  {
149  break;
150  }
151  ++m_handleIt;
152  }
153  if (!m_currentEntity)
154  {
155  // Every handle is supposed to be found as a typed variation in the entity access.
156  // If that unexpectedly fails, then the iterator is set to end.
157  m_handleIt = m_handleEnd;
158  }
159 }
