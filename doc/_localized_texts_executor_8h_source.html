171  typename LocalizedTextExecutor<LocalizedTextJob>::JobId jobId,

192  }
193

204  m_transactionCompleteCallback(transactionId, error);
205  std::lock_guard<std::mutex> const guard(m_mutex);
206  m_transactions.deleteTransaction(transactionId);
207  return;
208  }
209  }
210 
211  {

214  if (!jobDataPtr)

249 inline boost::optional<Utils::Error> LocalizedTextExecutor<LocalizedTextJob>::splitAndResubmit(
250  typename LocalizedTextExecutor<LocalizedTextJob>::TransactionId transactionId,
251  typename LocalizedTextExecutor<LocalizedTextJob>::JobId jobId)
252 {

254 
255  JobId const jobId1;
256  JobId const jobId2;
257  Execution::JobData oldJob;
258  DFL::Locale::LanguageTag language;
259 
260  const std::string failure = "Slicing of localization get text job failed. ";
261 
262  {
263  std::lock_guard<std::mutex> const guard(m_mutex);
264  oldJob = m_transactions.extractJobData(transactionId, jobId);
265  auto* transaction = m_transactions.transaction(transactionId);
266  if (!transaction)
267  {
268  return Utils::makeError(Errors::JobNotSliceable{}, failure + "(unknown transaction)");
269  }
270 
271  // prevent that the transaction can become empty (the old job is extracted already)
272  Execution::JobData placeholder;
273  placeholder.id = jobId1;
274  transaction->jobs[jobId1] = placeholder;
275  placeholder.id = jobId2;
276  transaction->jobs[jobId2] = placeholder;
277 
278  language = transaction->language;
279  }
280 
281  auto slices = sliceReferences(std::move(oldJob.references));
282  if (!slices)
283  {
284  return Utils::makeError(Errors::JobNotSliceable{}, failure + "(could not slice)");
285  }
286 
287  Execution::JobData jobData1;
288  Execution::JobData jobData2;
289  jobData1.id = jobId1;
290  jobData2.id = jobId2;
291 
292  auto filterRefs1 = makeFilters(slices->first, language);
293  auto filterRefs2 = makeFilters(slices->second, language);
294 
295  if (filterRefs1.size() != 1 || filterRefs2.size() != 1)
296  {
297  // each slice has to be representable with exactly one filter (because the old job was also)
298  return Utils::makeError(Errors::JobNotSliceable{}, failure + "(could not make filter)");
299  }
300 
301  jobData1.references = std::move(filterRefs1[0].references);
302  jobData1.filter = std::move(filterRefs1[0].filter);
303 
304  jobData2.references = std::move(filterRefs2[0].references);
305  jobData2.filter = std::move(filterRefs2[0].filter);
306 
307  setCallbacks(jobData1, transactionId, jobData1.id);
308  setCallbacks(jobData2, transactionId, jobData2.id);
309 
310  std::vector<JobDataNeededForSubmit> dataForSubmit;
311  {
312  std::lock_guard<std::mutex> const guard(m_mutex);
313  auto* transaction = m_transactions.transaction(transactionId);
314  if (!transaction)
315  {
316  return Utils::makeError(Errors::JobNotSliceable{}, failure + "(unknown transaction (2))");
317  }
318 
319  transaction->jobs[jobId1] = jobData1;
320  transaction->jobs[jobId2] = jobData2;
321 
322  JobDataNeededForSubmit const data1{jobData1.id, jobData1.handler, jobData1.createJobCallback};
323  JobDataNeededForSubmit const data2{jobData2.id, jobData2.handler, jobData2.createJobCallback};
324 
325  dataForSubmit.push_back(data1);
326  dataForSubmit.push_back(data2);
327  }
328 
329  submitTransaction(dataForSubmit, transactionId);
330  return boost::none;
