259  // 2 bit instance
260  // 10 bit component kind
261  // 10 bit free (0)

264  static const uint32_t InstanceBitAmount = 2;
265  static const uint32_t ComponentKindBitAmount = 10;
266  static const uint32_t ReservedFixBitAmount = 10;
267  static const uint32_t LicenseNumberBitAmount = 10;
268  static const uint32_t LicenseNumberBitOffset = 0;
269  static const uint32_t ReservedFixBitOffset = LicenseNumberBitOffset + LicenseNumberBitAmount;
270  static const uint32_t ComponentKindBitOffset = ReservedFixBitOffset + ReservedFixBitAmount;
271  static const uint32_t InstanceBitOffset = ComponentKindBitOffset + ComponentKindBitAmount;
272  static const uint32_t LicenseNumberBitMask = ((1U << LicenseNumberBitAmount) -1) << LicenseNumberBitOffset;
273  static const uint32_t ReservedFixBitMask = ((1U << ReservedFixBitAmount) -1) << ReservedFixBitOffset;
274  static const uint32_t ComponentKindBitMask = ((1U << ComponentKindBitAmount) -1) << ComponentKindBitOffset;
275  static const uint32_t InstanceBitMask = ((1U << InstanceBitAmount) -1) << InstanceBitOffset;
276  // When packed, the (ValidityTimer) counter is packed into the reserved bits.
277  // This keeps the packed size <= 8, which is important for the SIM card data, which shall be multiple of 8.
278  // PACKED License id bit layout
279  // 2 bit instance
280  // 10 bit component kind
281  // 8 bit counter <----
282  // 2 bit free (0)
283  // 10 bit license number
284  static const uint32_t PackedCounterBitAmount = 8;
285  // move bits upmost, to keep reserved bits directly above license number (this can be increased then)
286  static const uint32_t PackedCounterBitOffset = ReservedFixBitOffset + (LicenseNumberBitAmount - PackedCounterBitAmount);
287  static const uint32_t PackedCounterBitMask = ((1U << PackedCounterBitAmount) -1) << PackedCounterBitOffset;

294  // 3 bit validity

437 inline bool ProductLicense::unpack(const uint8_t * contents, uint8_t length)
438 {
439  m_id = (contents[0] << 24) | (contents[1] << 16) | (contents[2] << 8) | (contents[3]);
440  m_licData = (contents[4] << 8) | contents[5];
441  uint8_t const packSize = getPackedSize();
442  if (length < packSize)
443  {
444  return false;
445  }
446  if (packSize == PackedSizeValidityTimer)
447  {
448  m_dateDays = (contents[6] << 8) | contents[7];
449  // extract counter from reserved bits in m_id
450  uint32_t const counterBits = m_id & PackedCounterBitMask;
451  m_counter = static_cast<uint8_t>(counterBits >> PackedCounterBitOffset);
452  // clear counter bits in m_id
453  m_id &= ~PackedCounterBitMask;
454  }
455  else
456  {
457  m_dateDays = 0;
458  m_counter = 0;
459  }
460 
461  return true;
462 }

464 inline bool ProductLicense::pack(uint8_t * contents, uint8_t length) const
465 {
466  uint8_t const packSize = getPackedSize();
467  if (length != packSize)
468  {
469  return false;
470  }
471  uint32_t packedId = m_id;
472  if (packSize == PackedSizeValidityTimer)
473  {
474  contents[6] = static_cast<uint8_t>(m_dateDays >> 8);
475  contents[7] = static_cast<uint8_t>(m_dateDays);
476  // pack counter into reserved bits in m_id
477  packedId |= (m_counter << PackedCounterBitOffset);
478  }
479  contents[0] = static_cast<uint8_t>(packedId >> 24);
480  contents[1] = static_cast<uint8_t>(packedId >> 16);
481  contents[2] = static_cast<uint8_t>(packedId >> 8);
482  contents[3] = static_cast<uint8_t>(packedId);
483  contents[4] = static_cast<uint8_t>(m_licData >> 8);
484  contents[5] = static_cast<uint8_t>(m_licData);
485  return true;
486 }
