Detailed Description

Makes boost container streamable into DFL::Format::Stream.

TEST_F(TestBoostContainerIo, testFlatSet)
{
Stream str;
boost::container::flat_set<int> const empty;
str << empty;
EXPECT_STREQ("<>", str.cStr());
str.clear();
boost::container::flat_set<int> const s(m_arr, m_arr + m_sizeOfArr);
str << s;
EXPECT_STREQ("<0,1,10,100>", str.cStr());
}
TEST_F(TestBoostContainerIo, testFlatMap)
{
Stream str;
boost::container::flat_map<int, std::string> const empty;
str << empty;
EXPECT_STREQ("<>", str.cStr());
str.clear();
boost::container::flat_map<int, std::string> m;
m.insert(std::make_pair(0, std::string("null")));
m.insert(std::make_pair(1, std::string("one")));
m.insert(std::make_pair(2, std::string("two")));
str << m;
EXPECT_STREQ("<{0,null},{1,one},{2,two}>", str.cStr());
}
