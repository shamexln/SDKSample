Detailed Description

After construction DFL::Net::Adapter::Infos::get returns the list given to the constructor. On destruction the behaviour before construction is restored.

const Infos before{Infos::get()};
{
const MockInfos mock{
BuildInfo{}
.index(1)
.name("adapter_a")
.isLoopback(false)
.isMulticast(true)
.isConnected(true)
.addressesV4(
AddressesV4{AddressV4{"1.2.3.4"}, AddressV4{"11.22.33.44"}})
.macAddress(MacAddress{{1, 2, 3, 44, 55, 66}}),
BuildInfo{}
.index(22)
.name("adapter_b")
.isLoopback(true)
.isMulticast(false)
.isConnected(false)
.addressesV4(AddressesV4{AddressV4{"5.6.7.8"}})
.macAddress(MacAddress{{5, 6, 7, 88, 99, 111}})};
const Infos adapters{Infos::get()};
EXPECT_NE(adapters, before);
const Info a{adapters.findByName("adapter_a").value()};
EXPECT_EQ(a.index(), 1);
EXPECT_EQ(a.name(), "adapter_a");
EXPECT_FALSE(a.isLoopback());
EXPECT_TRUE(a.isMulticast());
EXPECT_TRUE(a.isConnected());
EXPECT_EQ(
a.addressesV4(),
(AddressesV4{AddressV4{"1.2.3.4"}, AddressV4{"11.22.33.44"}}));
EXPECT_EQ(a.macAddress(), (MacAddress{{1, 2, 3, 44, 55, 66}}));
const Info b{adapters.findByName("adapter_b").value()};
EXPECT_EQ(b.index(), 22);
EXPECT_EQ(b.name(), "adapter_b");
EXPECT_TRUE(b.isLoopback());
EXPECT_FALSE(b.isMulticast());
EXPECT_FALSE(b.isConnected());
EXPECT_EQ(b.addressesV4(), (AddressesV4{AddressV4{"5.6.7.8"}}));
EXPECT_EQ(b.macAddress(), (MacAddress{{5, 6, 7, 88, 99, 111}}));
}
const Infos after{Infos::get()};
EXPECT_EQ(after, before);
