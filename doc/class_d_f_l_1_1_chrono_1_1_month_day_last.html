Detailed Description

This represents the last day of a month.

An instance can be constructed from a DFL::Chrono::Month or using the aug/last syntax.

{
const MonthDayLast mdl(aug);
ASSERT_EQ(aug, mdl.month());
ASSERT_TRUE(mdl.ok());
}
{
const MonthDayLast mdl(Month(13));
ASSERT_FALSE(mdl.ok());
}
{
ASSERT_EQ( aug/last, MonthDayLast(Month(8)));
}
