216  template <class State, class Entity>
217  AddToChangeSetResult addToChangeSet(const DFL::NotNull<DFL::Mdib::SharedPtr<State>>& state,
218  typename std::enable_if<std::is_base_of<DFL::Mdib::AbstractContextState, State>::value>::type* = nullptr)
219  {
220  const auto handle = DFL::Mdib::BasicHandle<Entity>(m_state->contextStateHandle);
221  const auto descriptorHandle = DFL::Mdib::BasicHandle<typename Entity::ParentType>(m_state->descriptorHandle);
222  if (!handle.isValid())
223  {
224  S31_THROW_EXCEPTION_WITH_ERROR(Errors::IncomingDataNotUnderstood{}, "ContextState with invalid handle " << handle);
225  }
226 
227  // update existing state?
228  const auto previousState = m_mdib.entity(handle);
229  if (previousState)
230  {
231  if (m_state->stateVersion <= previousState->stateVersion())
232  {
233  // known state version
234  S31_STREAM_TRACE_LOGGER("Sdc", "ContextState ", handle,
235  " in version already known. Received: ", m_state->stateVersion,
236  ", already have: ", previousState->stateVersion());
237  return AddToChangeSetResult::Handled;
238  }
239  }
240 
241  const auto indicator = m_mdib.entity(descriptorHandle);
242  if (!indicator)
243  {
244  S31_THROW_EXCEPTION_WITH_ERROR(Errors::IncomingDataNotUnderstood{}, "ContextState " << handle << " has no descriptor " << descriptorHandle);
245  }
246  if (m_state->descriptorVersion != indicator->descriptorVersion())
247  {
248  S31_THROW_EXCEPTION_WITH_ERROR(Errors::StateDescriptorVersionMismatch{},
249  "Context state " << handle
250  << " update/insertion received whose descriptor ("
251  << indicator->handle() <<") version is invalid. "
252  << "Received: " << m_state->descriptorVersion
253  << ", expected: " << indicator->descriptorVersion() << ".");
254  }
255 
256  auto newContextState = DFL::asNotNull(DFL_MDIB_MAKE_SHARED<Entity>(
257  handle,
258  state,
259  m_state->stateVersion,
260  descriptorHandle));
261 
262  if (newContextState->state()->contextAssociation() == DFL::Mdib::ContextAssociation::NotAssociated)
263  {
264  m_changeSet.addOneToRemove(newContextState);
265  return AddToChangeSetResult::Handled;
266  }
267 
268  if (previousState)
269  {
270  m_changeSet.addOneToUpdate(newContextState);
271  }
272  else
273  {
274  m_changeSet.addOneToInsert(newContextState);
275  }
276 
277  return AddToChangeSetResult::Handled;
278  }

280  template <class State, class Entity>
281  AddToChangeSetResult addToChangeSet(const DFL::NotNull<DFL::Mdib::SharedPtr<State>>& state,
282  typename std::enable_if<!std::is_base_of<DFL::Mdib::AbstractContextState, State>::value>::type* = nullptr)
283  {
284  auto oldEntity = m_mdib.entity(DFL::Mdib::BasicHandle<Entity>(m_state->descriptorHandle));
285  if (!oldEntity)
286  {
287  return AddToChangeSetResult::NotHandled;
288  }
289 
290  if (m_state->stateVersion <= oldEntity->stateVersion())
291  {
292  S31_STREAM_TRACE_LOGGER("Sdc", "Skip ", oldEntity->handle(), ". No newer version. Received: ", m_state->stateVersion, ", already have: ", oldEntity->stateVersion());
293  return AddToChangeSetResult::Handled;
294  }
295  if (m_state->descriptorVersion != oldEntity->descriptorVersion())
296  {
297  S31_THROW_EXCEPTION_WITH_ERROR(Errors::StateDescriptorVersionMismatch{},
298  "State update received for " << oldEntity->handle()
299  << " with invalid descriptor version. "
300  << "Received: " << m_state->descriptorVersion
301  << ", expected: " << oldEntity->descriptorVersion() << ".");
302  }
303 
304  auto newEntity = DFL_MDIB_MAKE_SHARED<Entity>(
305  oldEntity->handle(), oldEntity->descriptor(), oldEntity->descriptorVersion(),
306  state, m_state->stateVersion, oldEntity->parent());
307 
308  m_changeSet.addOneToUpdate(DFL::asNotNull(newEntity));
309  S31_STREAM_TRACE_LOGGER("Sdc", "Update state ", oldEntity->handle());
310  return AddToChangeSetResult::Handled;
311  }

629  template <typename Descriptor>
630  void handleContextStates(const DFL::NotNull<DFL::Mdib::SharedPtr<Descriptor>>& /*d*/)
631  {
632  using Indicator = typename EntityTypeOfDescriptor<Descriptor>::type;
633  using Context = typename ContextOfContextIndicator<Indicator>::type;
634  using StatePtr = DFL::NotNull<DFL::Mdib::SharedPtr<typename Context::StateType>>;
635 
636  // range states
637  auto stateRange = m_modifications.stateIndex().equal_range(m_descriptor->handle);
638  for (auto stateIt = stateRange.first; stateIt != stateRange.second; ++stateIt)
639  {
640  const auto& stateReport = m_modifications.states()[stateIt->second];
641  const auto handle = DFL::Mdib::BasicHandle<Context>(stateReport.contextStateHandle);
642  if (!handle.isValid())
643  {
644  S31_THROW_EXCEPTION_WITH_ERROR(Errors::IncomingDataNotUnderstood{},
645  "Context state handle '" << handle << "' is not valid.");
646  }
647  const StatePtr* state = std::get_if<StatePtr>(&stateReport.element);
648  if (!state)
649  {
650  S31_THROW_EXCEPTION_WITH_ERROR(Errors::IncomingDataNotUnderstood{},
651  "Context state " << handle << " is not of suiting type for " << m_descriptor->handle);
652  }
653 
654  // update existing entity?
655  const auto previousEntity = m_mdib.entity(handle);
656  if (previousEntity && stateReport.stateVersion <= previousEntity->stateVersion())
657  {
658  S31_STREAM_TRACE_LOGGER("Sdc", "ContextState with handle ", handle, " in version already known");
659  continue;
660  }
661 
662  if (m_descriptor->version != stateReport.descriptorVersion)
663  {
664  S31_THROW_EXCEPTION_WITH_ERROR(Errors::IncomingDataNotUnderstood{},
665  "Descriptor report (" << m_descriptor->changeType << ") for "
666  << m_descriptor->handle << " has a context state " << handle
667  << " with a non-matching descriptor version. Descriptor: "
668  << m_descriptor->version << ", state's descriptor: " << stateReport.descriptorVersion);
669  }
670 
671  auto entity = DFL_MDIB_MAKE_SHARED<Context>(
672  handle,
673  *state,
674  stateReport.stateVersion,
675  DFL::Mdib::BasicHandle<Indicator>(m_descriptor->handle));
676 
677  if (entity->state()->contextAssociation() == DFL::Mdib::ContextAssociation::NotAssociated)
678  {
679  m_changeSet.addOneToRemove(DFL::asNotNull(entity));
680  continue;
681  }

714  m_report(report)

716 
717  DFL::Mdib::MdibChangeSet build()
718  {
719  DFL::Mdib::MdibChangeSet result;
720 
721  if (!checkMdibVersion(m_mdib, m_report))
722  {
723  return result;
724  }
725 
726  if (m_report.elements.empty())
727  {
728  return result;
729  }
730 
731  Inserter inserter(result, m_mdib, *m_report.elements.begin(), m_report);
732  for (const auto& reportPart : m_report.elements)
733  {
734  inserter.setCurrentReportPart(reportPart);
735  std::visit(inserter, reportPart.element);
736  }
737 
738  return result;
739  }
740 
741  private:
745  template <typename Mod>
746  struct InserterOfModification
747  {
748  using AllModifications = boost::mpl::vector<DescriptorModifications, StateModifications>;
749  using AllInserters = boost::mpl::vector<ChangeSetInserterDescriptor, ChangeSetInserterState>;
750  using iter = typename boost::mpl::find_if<AllModifications, std::is_same<boost::mpl::_1, Mod>>::type;
751  using type = typename boost::mpl::at<AllInserters, boost::mpl::int_<iter::pos::value>>::type;
752  };
753  using Inserter = typename InserterOfModification<Modifications>::type;
754 
755  bool checkMdibVersion(const DFL::Mdib::Mdib& mdib, const Modifications& modifications)
756  {
757  const auto& mdibVersion = mdib.mdibVersion();
758  const auto& reportVersion = modifications.mdibVersion;
759 
760  if (mdibVersion.isOlderVersionOf(reportVersion) || mdibVersion == reportVersion)
761  {
762  return true;
763  }
764 
765  if (mdibVersion.isOtherSequenceOf(reportVersion))
766  {
767  S31_THROW_EXCEPTION_WITH_ERROR(Errors::MdibSequenceChanged{},
768  "New MDIB sequence detected. Cannot handle report. old: "
769  << mdibVersion << " new: " << reportVersion);
770  }
771 
772  S31_STREAM_TRACE_LOGGER("Sdc",
773  "Received Mdib version is an older version than existing. Have ", mdibVersion, ", received ", reportVersion);
774  return false;
775  }
