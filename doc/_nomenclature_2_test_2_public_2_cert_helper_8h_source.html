320 inline X509RevocationLists createCertificateRevocation(const DFL::Crypto::Certificate& cert)
321 {
322  // Note: this function should be removed when DFL::Crypto::Test implements utility to
323  // create certificate revocation lists for testing. The rootCaCertKey, and rootCaCertString
324  // are expecting to be equal to the test root Ca cert and key from DFL::Crypto
325 
326  typedef std::unique_ptr<::ASN1_INTEGER, decltype(&::ASN1_INTEGER_free)> Asn1IntegerPtr;
327  typedef std::unique_ptr<::X509_CRL , decltype(&::X509_CRL_free) > X509CrlPtr;
328  typedef std::unique_ptr<::X509_REVOKED, decltype(&::X509_REVOKED_free)> X509RevokedPtr;
329 
330  using namespace DFL::Crypto;
331 
332  auto caKey = PrivateKey(PemString(S31::Nomenclature::Test::Impl::rootCaCertKey));
333  auto caCert = Certificate(PemString(S31::Nomenclature::Test::Impl::rootCaCertString));
334 
335  auto crl = X509CrlPtr(::X509_CRL_new(), ::X509_CRL_free);
336 
337  if (!X509_CRL_set_version(crl.get(), 1))
338  {
339  S31_THROW_EXCEPTION(InternalConsistencyFailure, "Cannot set crl version");
340  }
341 
342  auto tm = OpenSsl::Asn1TimeNew();
343  if (!tm)
344  {
345  S31_THROW_EXCEPTION(InternalConsistencyFailure, "Cannot create time for crl");
346  }
347  X509_gmtime_adj(tm.get(), 0);
348  X509_CRL_set_lastUpdate(crl.get(), tm.get());
349 
350  static int crlSerial = 1;
351  auto serial = Asn1IntegerPtr(::ASN1_INTEGER_new(), ::ASN1_INTEGER_free);
352  ASN1_INTEGER_set(serial.get(), crlSerial++);
353  X509_CRL_add1_ext_i2d(crl.get(), NID_crl_number, serial.get(), 0, 0);
354 
355  X509_gmtime_adj(tm.get(), 24 * 60 * 60);
356  X509_CRL_set_nextUpdate(crl.get(), tm.get());
357 
358  if (!X509_CRL_set_issuer_name(crl.get(), X509_get_subject_name(caCert.certificate())))
359  {
360  S31_THROW_EXCEPTION(InternalConsistencyFailure, "Unable to set the issuer name for the new CRL");
361  }
362 
363  auto revoked = X509RevokedPtr(::X509_REVOKED_new(), ::X509_REVOKED_free);
364  if (!revoked)
365  {
366  S31_THROW_EXCEPTION(InternalConsistencyFailure, "Unable to create an X509_REVOKED object");
367  }
368 
369  auto* certX509 = const_cast<::X509*>(cert.certificate());
370  if (!X509_REVOKED_set_serialNumber(revoked.get(), X509_get_serialNumber(certX509)))
371  {
372  S31_THROW_EXCEPTION(InternalConsistencyFailure, "Unable to set serial number");
373  }
374 
375  // cert was one second ago revoked
376  X509_gmtime_adj(tm.get(), -1);
377  if (!X509_REVOKED_set_revocationDate(revoked.get(), tm.get()))
378  {
379  S31_THROW_EXCEPTION(InternalConsistencyFailure, "Unable to set revocation date");
380  }
381 
382  X509_CRL_add0_revoked(crl.get(), revoked.release());
383 
384  X509_CRL_sort(crl.get());
385  // const_cast: OpenSSL is not entirely const-correct and DFL::Crypto doesn't want to expose a non-const pointer to EVP_PKEY
386  X509_CRL_sign(crl.get(), const_cast<EVP_PKEY*>(caKey.privateKey()), EVP_sha1());
387 
388  return X509RevocationLists{ std::move(crl) };
389 }
