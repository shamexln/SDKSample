87 int addExtension(X509* cert, int nid, const char* value)
88 {
89  X509V3_CTX ctx;
90 
91  // This sets the 'context' of the extensions.
92  // No configuration database
93  X509V3_set_ctx_nodb(&ctx);
94 
95  // Issuer and subject certs: both the target since it is self-signed, no
96  // request and no CRL
97  X509V3_set_ctx(&ctx, cert, cert, nullptr, nullptr, 0);
98 
99  auto extension(std::shared_ptr<X509_EXTENSION>(X509V3_EXT_conf_nid(nullptr, &ctx, nid, const_cast<char*>(value)), X509_EXTENSION_free));
100  THROW_IF_NOT(extension);
101 
102  X509_add_ext(cert, extension.get(), -1);
103  return 1;
104 }
