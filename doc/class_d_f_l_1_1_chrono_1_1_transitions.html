Detailed Description

Models a container holding transitions from std to dst and vice versa for a specific time zone in a specific year-range.

const TimeZone tzWithoutDst(IanaFormat("STD+01"));
const TimeZone tzWithDst (IanaFormat("CET+01:00CEST+02:00,M03.5.0/02:00,M10.5.0/03:00"));
const Transitions actual(calcTransitions(tzWithDst, Year(2000), Years(5)));
const Transitions expected = boost::assign::list_of
(Transition(SystemDays(Year(2000)/mar/26) + Hours(1), Hours(2), "CEST", true ))
(Transition(SystemDays(Year(2000)/oct/29) + Hours(1), Hours(1), "CET" , false))
(Transition(SystemDays(Year(2001)/mar/25) + Hours(1), Hours(2), "CEST", true ))
(Transition(SystemDays(Year(2001)/oct/28) + Hours(1), Hours(1), "CET" , false))
(Transition(SystemDays(Year(2002)/mar/31) + Hours(1), Hours(2), "CEST", true ))
(Transition(SystemDays(Year(2002)/oct/27) + Hours(1), Hours(1), "CET" , false))
(Transition(SystemDays(Year(2003)/mar/30) + Hours(1), Hours(2), "CEST", true ))
(Transition(SystemDays(Year(2003)/oct/26) + Hours(1), Hours(1), "CET" , false))
(Transition(SystemDays(Year(2004)/mar/28) + Hours(1), Hours(2), "CEST", true ))
(Transition(SystemDays(Year(2004)/oct/31) + Hours(1), Hours(1), "CET" , false))
(Transition(SystemDays(Year(2005)/mar/27) + Hours(1), Hours(2), "CEST", true ))
(Transition(SystemDays(Year(2005)/oct/30) + Hours(1), Hours(1), "CET" , false))
;
ASSERT_EQ(expected, actual);

Instances of this class are typically constructed by a factory function e.g. DFL::Chrono::calcTransitions.

Access to elements of this container are gained via an iterator interface. Thus making even complex tasks like getting a string representation of all transitions in a specific year in local time trivial:

namespace
{
LocalTime transitionToLocalTime(const Transition& t)
{
return ZonedTime(t.timePoint(), tzWithDst).localTime();
}
}
TEST(Transitions, iteratorInterface)
{
std::ostringstream str;
str << "|";
std::vector<LocalTime> localTimePoints;
const Transitions a(calcTransitions(tzWithDst, Year(2017)));
localTimePoints.reserve(a.size());
std::transform(a.cbegin()
, a.cend()
, std::back_inserter(localTimePoints)
, transitionToLocalTime
);
std::copy(localTimePoints.begin()
, localTimePoints.end()
, std::ostream_iterator<LocalTime>(str, "|")
);
// these local times points are the first one
// that are in the "new" std/dst range
// these are not the time point on which you think that
// the clock has to be changed +/-1 hour.
ASSERT_EQ(std::string("|2017-03-26T03:00:00|2017-10-29T02:00:00|")
, str.str());
}
