156 template<class T_Builder>
157 inline void ParticipantAlertMapper::mapGenericAlertConditionStateFromGsoap(T_Builder& dest,
158  const pm__AlertConditionState& src)
159 {
160  dest.activationState(mapAlertActivationFromGsoap(src.ActivationState));
161  if (src.ActualPriority)
162  {
163  dest.actualPriority(mapAlertConditionPriorityFromGsoap(*src.ActualPriority));
164  }
165  if (src.Presence)
166  {
167  if (src.Presence == "true" || src.Presence == "1")
168  {
169  dest.presence(DFL::Mdib::AlertConditionPresence::Present);
170  }
171  else if (src.Presence == "false" || src.Presence == "0")
172  {
173  dest.presence(DFL::Mdib::AlertConditionPresence::NotPresent);
174  }
175  else
176  {
177  // invalid xsd:boolean value
178  S31_THROW_EXCEPTION(ExternalDataMappingFailed, "invalid xsd:boolean value: " << *src.Presence);
179  }
180  }
181  else
182  {
183  // The implied value SHALL be "false".
184  dest.presence(DFL::Mdib::AlertConditionPresence::NotPresent);
185  }
186 
187  if (src.Rank)
188  {
189  dest.rank(DFL::Mdib::Rank(*src.Rank));
190  }
191  else
192  {
193  dest.rank(boost::none);
194  }
195 
196  if (src.DeterminationTime)
197  {
198  dest.determinationTime(
199  DFL::Mdib::Convert::fromBicepsTimestamp(DFL::Mdib::BicepsTimestamp{*src.DeterminationTime}));
200  }
201  else
202  {
203  dest.determinationTime(boost::none);
204  }
205  if (src.ActualConditionGenerationDelay)
206  {
207  dest.actualConditionGenerationDelay(ParticipantBaseMapper::toMdibDuration(*src.ActualConditionGenerationDelay));
208  }
209 
210  dest.extensions(ParticipantBaseMapper::mapExtensionsFromGsoap(src.ext__Extension));
211 }
