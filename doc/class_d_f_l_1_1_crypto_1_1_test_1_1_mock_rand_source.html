Detailed Description

Exchanges the randsource of DFL::Crypto::RandomDevice with one that always returns the given value.

RandomDevice const rd{};
// produces random numbers
// ...
{
const RandomDeviceResultType zero{0};
const MockRandSource mockZero{};
// ... now it will always produce the same number
ASSERT_EQ(rd(), zero);
{
const RandomDeviceResultType one{1};
const MockRandSource mockOne{one};
ASSERT_EQ(rd(), one);
}
ASSERT_EQ(rd(), zero);
}
// produces random numbers again ...
