Detailed Description

Provides encryption/decryption of bytes based on AES-128 in GCM mode.

  • For details on Advanced Encryption Standard (AES), see https://en.wikipedia.org/wiki/Advanced_Encryption_Standard
  • For details on Galois/Counter Mode (GCM), see https://en.wikipedia.org/wiki/Galois/Counter_Mode
using namespace std::string_view_literals;
using ::DFL::Crypto::Cipher::decrypt;
using ::DFL::Crypto::Cipher::encrypt;
using ::DFL::Crypto::Cipher::Symmetric::Aes128Gcm;
using ::DFL::Encoding::Convert::Bytes::asString;
const auto key{"0123456789012345"sv};
const auto iv{"012345678901"sv};
auto aes{Aes128Gcm{key, iv}};
{
const auto message{"The quick brown fox jumps over the lazy dog"sv};
auto& enc{aes.encrypter()};
const auto encrypted{encrypt(enc, message)};
const auto tag{enc.tag()};
auto& dec{aes.decrypter()};
dec.tag(tag);
const auto decrypted{decrypt(dec, encrypted)};
EXPECT_EQ(message, asString(decrypted));
}
// cipher can be reused
{
const auto message{"dog lazy the over jumps fox brown quick The"sv};
auto& enc{aes.encrypter()};
const auto encrypted{encrypt(enc, message)};
const auto tag{enc.tag()};
auto& dec{aes.decrypter()};
dec.tag(tag);
const auto decrypted{decrypt(dec, encrypted)};
EXPECT_EQ(message, asString(decrypted));
}
