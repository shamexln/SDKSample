Detailed Description

Interface to be implemented by the concrete persistence service.

An instance of this interface can then be injected into this lib via DFL::Password::Config::Registry::persistence.

Returns true on success and false on any error.

std::unique_ptr<Persistence> const p = std::make_unique<VolatilePersistence>();
EXPECT_TRUE(p->save(bytes, Persistence::Slot::Level2));
EXPECT_TRUE(p->save(bytes, Persistence::Slot::Level3));
EXPECT_TRUE(p->reset(Persistence::Slot::Level2));
EXPECT_TRUE(p->load(Persistence::Slot::Level2).empty());
EXPECT_EQ(p->load(Persistence::Slot::Level3), bytes);

EXPECT_TRUE(p->load(Persistence::Slot::Level2).empty());
EXPECT_TRUE(p->load(Persistence::Slot::Level3).empty());
