boost::optional< ExtensionType_T > DFL::Mdib::Ext::ParticipantModelExtensions::get ( const Element_T &  element)
static

Searches for the requested known extension in the extension of passed (MDIB state or descriptor) type.

In case exactly one of the generic extension match the requested known extension, it will be returned. Otherwise boost::none is returned.

An example usage for AlertSignalFlashing:

// Device side usage: create the AlertSignalFlashing instance and add it
// after conversion with the free function
// DFL::Mdib::Ext::ParticipantModelExtensions::toExtension to any extendable
// MDIB class
const AlertSignalFlashing deviceAsf{
{DFL::Mdib::Handle{"handle1"}, DFL::Mdib::Handle{"handle2"}}};
DFL::Mdib::AlertSystemState const ass{
DFL::Mdib::AlertActivation::On,
boost::none,
boost::none,
{},
{},
// just add the extension in the ctor an MDIB class
{DFL::Mdib::Ext::ParticipantModelExtensions::toExtension(deviceAsf)}};
// Client side usage: use the free function
// DFL::Mdib::Ext::ParticipantModelExtensions::get<AlertSignalFlashing>() to
// retrieve the AlertSignalFlashing from any state or other extendable
// element from the MDIB.
if (const auto clientAsf =
ParticipantModelExtensions::get<AlertSignalFlashing>(ass))
{
EXPECT_THAT(clientAsf->flashingSignals(), SizeIs(2));
EXPECT_THAT(clientAsf->flashingSignals(),
Eq(deviceAsf.flashingSignals()));
}

template<class ExtensionType_T , class Element_T >
std::vector< ExtensionType_T > DFL::Mdib::Ext::ParticipantModelExtensions::getAll ( const Element_T &  element)

Searches for the requested known extension in the extension of passed (MDIB state or descriptor) type.

All of the generic extensions that match the requested known extension will be returned.

An example usage for InvocationPerformedFrom:

// Device side usage: create the InvocationPerformedFrom instances and add
// them after conversion with the free function
// DFL::Mdib::Ext::ParticipantModelExtensions::toExtension to any extendable
// MDIB class
const InvocationPerformedFrom immediate{
DFL::Mdib::Ext::Vicinity::ImmediateVicinity};
const InvocationPerformedFrom close{
DFL::Mdib::Ext::Vicinity::CloseVicinity};
DFL::Mdib::ActivateOperationDescriptor const aod{
boost::none,
boost::none,
DFL::Mdib::Handle{"targetHandle"},
boost::none,
boost::none,
DFL::Mdib::Retriggerable::RetriggerLeavesEffectiveTimeoutUntouched,
DFL::Mdib::AccessLevel::Other,
{},
// just add the extension in the ctor an MDIB class
{DFL::Mdib::Ext::ParticipantModelExtensions::toExtension(immediate),
DFL::Mdib::Ext::ParticipantModelExtensions::toExtension(close)}};
// Client side usage: use the free function
// DFL::Mdib::Ext::ParticipantModelExtensions::getAll<InvocationPerformedFrom>()
// to retrieve the InvocationPerformedFrom from any state or other
// extendable element from the MDIB.
const auto clientIpf =
ParticipantModelExtensions::getAll<InvocationPerformedFrom>(aod);
EXPECT_THAT(clientIpf, SizeIs(2u));
for (const auto& ipf : clientIpf)
{
EXPECT_THAT(ipf.vicinity(), Ne(DFL::Mdib::Ext::Vicinity::Any));
}
