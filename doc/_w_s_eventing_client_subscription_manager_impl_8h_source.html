218 template<class Jobs>
219 int WSEventingClientSubscriptionManagerImpl<Jobs>::dispatch(struct soap* soapHandle)
220 {
221  const auto* action = S31::Gsoap::soapAction(soapHandle);
222  if (!action)
223  {
224  // soap header 'action' is required element
225  return soapHandle->error = SOAP_NO_METHOD;
226  }
227  if (auto res = SoapUtils::checkThatPeerHasRoles(soapHandle, DFL::Oids{Nomenclature::SdcServiceProvider}))
228  {
229  return res;
230  }
231 
232  const auto identifier(SoapHeaderUtils::getSubscriptionIdentifier(soapHandle));
233 
234  if (identifier.empty())
235  {
236  S31_STREAM_TRACE_LOGGER(
237  "Soap.WS", "Client: WS-Eventing: no subscription identifier found for action '", action, "'");
238  return SOAP_NO_METHOD;
239  }
240 
241  // First check for general WS-Eventing messages for the client side (SubscriptionEnd)
242  if (StackConstants::WS_EVENTING_SUBSCRIPTIONEND_ACTION == action)
243  {
244  EventSinkService<WSEventingClientSubscriptionManagerImpl<Jobs>> service(soapHandle, this->shared_from_this(), identifier);
245  int const r = service.dispatch();
246  if (r != SOAP_OK && r != SOAP_STOP)
247  {
248  // SOAP_STOP is fine, since for one way messages (SubscriptionEnd in this case)
249  // the HTTP status "202 Accepted" results in SOAP_STOP (code 1000)
250  S31_STREAM_WARN_LOGGER("Soap.WS", "Dispatching of SubscriptionEnd failed, error code: ", Gsoap::soapCode(r));
251  }
252  return r;
253  }
254 
255  std::shared_ptr<IWSEventingClientListener> callback;
256  {
257  std::lock_guard<std::mutex> const guard(m_mutex);
258  if (auto subscription = findSubscriptionNoLock(WSEventingClientSubscriptionId(identifier)))
259  {
260  if (subscription->isSubscribedToAction(action))
261  {
262  callback = subscription->getCallback();
263  }
264  }
265  else
266  {
267  DFL::Format::Stream msg;
268  msg << "Client: WS-Eventing: unknown subscription identifier " << identifier;
269  S31_STREAM_WARN_LOGGER("Soap.WS", msg.cStr());
270  return SoapHeaderUtils::prepareSoapSenderFault(soapHandle, msg.cStr(), nullptr);
271  }
272  }
273 
274  if (callback)
275  {
276  int r = 0;
277  try
278  {
279  r = callback->dispatchEvent(soapHandle);
280  }
281  catch (const std::exception& e)
282  {
283  S31_STREAM_ERROR_LOGGER(
284  "Soap.WS",
285  "Caught exception: '",
286  e.what(),
287  "' while dispatching event with soap action '",
288  action,
289  "'");
290  }
291  if (r && r != SOAP_STOP)
292  {
293  S31_STREAM_TRACE_LOGGER("Soap.WS", "Client: WS-Eventing: dispatching event to client failed with error ", r);
294  }
295  return r;
296  }
297 
298  S31_STREAM_TRACE_LOGGER(
299  "Soap.WS", "Client: WS-Eventing: no callback for subscription ", identifier, " and action ", action);
300 
301  return SOAP_NO_METHOD;
302 }

402  if (!eprPtr)

409  !reason.empty() ? DFL::Locale::Utf8(", reason: ") + reason : DFL::Locale::Utf8(""));
410 
411  auto job = std::make_unique<typename Jobs::Unsubscribe>(m_handlerOfUnsubscribeJobs, subscriptionId, *eprPtr);
412  if (auto queue = m_outgoingJobQueue.lock())
413  {
414  queue->addAsyncJob(DFL::asNotNull(std::move(job)));
