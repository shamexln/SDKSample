Detailed Description

Provides all methods concerning the expiration of security codes.

EXPECT_TRUE(env.m_expiryAction.save(expireInOneMonth));
EXPECT_EQ(env.m_expiryAction.load(), expireInOneMonth);
systemMockClock.setClock(expirationDay - Weeks{2});
EXPECT_FALSE(env.m_expiryAction.isExpired());
EXPECT_TRUE(env.m_expiryAction.willExpireSoon());
systemMockClock.setClock(expirationDay);
EXPECT_TRUE(env.m_expiryAction.isExpired());
EXPECT_TRUE(env.m_expiryAction.willExpireSoon());
EXPECT_TRUE(env.m_expiryAction.save(boost::none));
EXPECT_FALSE(env.m_expiryAction.isExpired());
EXPECT_FALSE(env.m_expiryAction.willExpireSoon());
