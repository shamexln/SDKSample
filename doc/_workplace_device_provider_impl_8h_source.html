564  // Intended to be called while the caller holds the mutex
565  bool remoteLocationIsGood(const DFL::Mdib::Mdib& mdib, std::string& notGoodInfo)
566  {
567  if (!m_ownLocation)
568  {
569  notGoodInfo = "Own location is not set.";
570  return false;
571  }
572 
573  const auto range = mdib.range<DFL::Mdib::LocationContext>();
574  if (range.empty())
575  {
576  notGoodInfo = "Remote Mdib does not contain location contexts.";
577  return false;
578  }
579 
580  auto associatedContextIt = std::find_if(range.begin(), range.end(), [&](const DFL::NotNull<DFL::Mdib::LocationContextSharedPtr>& context)
581  {
582  return context->state()->contextAssociation() == DFL::Mdib::ContextAssociation::Associated;
583  });
584 
585  if (associatedContextIt == range.end())
586  {
587  notGoodInfo = "Remote Mdib does not contain a associated location context.";
588  return false;
589  }
590 
591  const auto& state = (*associatedContextIt)->state();
592  const auto& identifications = state->identifications();
593 
594  if (!std::any_of(identifications.begin(), identifications.end(),
595  [this](const auto& identifier)
596  {
597  return DFL::Mdib::referencesSameInstance(identifier, m_ownLocation->identifier) == DFL::Mdib::ReferencesSameInstance::True;
598  }))
599  {
600  notGoodInfo = "None of the identifiers of the associated location context in the remote Mdib matches.";
601  return false;
602  }
603 
604  if (state->validators().empty())
605  {
606  notGoodInfo = "The associated location context in the remote Mdib has no validator.";
607  return false;
608  }
609  return true;
610  }

612  // Intended to be called while the caller holds the mutex
613  bool remoteMdsIsGood(const DFL::Mdib::Mdib& mdib, std::string& notGoodInfo) const
614  {
615  const auto range = mdib.range<DFL::Mdib::Mds>();
616  if (range.empty())
617  {
618  notGoodInfo = "Remote Mdib does not contain a Mds.";
619  return false;
620  }
621  if (++range.begin() != range.end())
622  {
623  notGoodInfo = "Remote Mdib contains more than one Mds.";
624  return false;
625  }
626 
627  const auto mds = *range.begin();
628  const auto type = mds->descriptor()->type();
629  if (!type)
630  {
631  notGoodInfo = "Remote Mdib's Mds has no type.";
632  return false;
633  }
634 
635  if (m_ownMdsType && DFL::Mdib::isSemanticallyEqual(*m_ownMdsType, *type))
636  {
637  notGoodInfo = "Remote Mdib's Mds type is the same as ours.";
638  return false;
639  }
