Detailed Description

Contains utilities for making assertion which fail at compile time.

These assert macros take compile time expression which lead to either "true" or "false". If a message is provided it can be found in the compiler output if the given expression leads to a compile time "false".

The assert macros can be used in various scopes (namespaces, private sections, functions, ...).

// all these tests should ensure that the static assert macros compile
namespace
{
// Namespace scope
DFL_STATIC_ASSERT(sizeof(int) >= sizeof(short));
DFL_STATIC_ASSERT(sizeof(char) == 1);
DFL_STATIC_ASSERT_MSG(sizeof(int) >= sizeof(short), "msg1");
DFL_STATIC_ASSERT_MSG(sizeof(char) == 1, "msg2");
// Function (block) scope
void f()
{
DFL_STATIC_ASSERT(sizeof(int) >= sizeof(short));
DFL_STATIC_ASSERT(sizeof(char) == 1);
DFL_STATIC_ASSERT_MSG(sizeof(int) >= sizeof(short), "msg3");
DFL_STATIC_ASSERT_MSG(sizeof(char) == 1, "msg4");
}
struct Bob
{
private: // can be in private, to avoid namespace pollution
DFL_STATIC_ASSERT(sizeof(int) >= sizeof(short));
DFL_STATIC_ASSERT(sizeof(char) == 1);
DFL_STATIC_ASSERT_MSG(sizeof(int) >= sizeof(short), "msg5");
DFL_STATIC_ASSERT_MSG(sizeof(char) == 1, "msg6");
public:
// Member function scope: provides access to member variables
int x;
char c;
int f()
{
DFL_STATIC_ASSERT(sizeof(x) >= sizeof(short));
DFL_STATIC_ASSERT(sizeof(c) == 1);
DFL_STATIC_ASSERT_MSG(sizeof(x) >= sizeof(short), "msg7");
DFL_STATIC_ASSERT_MSG(sizeof(c) == 1, "msg8");
return x;
}
};
// Template class scope
template <class Int, class Char>
struct Bill
{
BOOST_STATIC_CONSTANT(int, value = 1);
private: // can be in private, to avoid namespace pollution
DFL_STATIC_ASSERT(sizeof(Int) > sizeof(Char));
DFL_STATIC_ASSERT_MSG(sizeof(Int) > sizeof(Char), "msg9");
public:
// Template member function scope: provides access to member variables
Int x;
Char c;
template <class Int2, class Char2>
void f(Int2 , Char2 )
{
DFL_STATIC_ASSERT(sizeof(Int) == sizeof(Int2));
DFL_STATIC_ASSERT(sizeof(Char) == sizeof(Char2));
DFL_STATIC_ASSERT_MSG(sizeof(Int) == sizeof(Int2), "msg10");
DFL_STATIC_ASSERT_MSG(sizeof(Char) == sizeof(Char2), "msg11");
}
};
}

For being able to compare enum values of different enum types without generating warnings a special macro has been implemented which casts boths values to int and then make the comparison.

using Nums = enum { Zero, One, Two };
using Zahlen = enum { Null, Eins, Zwei };
// this line should not trigger a warning
// since we compile with warn as error
// that requirement will not slip
DFL_STATIC_ASSERT_INCOMPATIBLE_ENUMS_EQ(Zero , Null);
DFL_STATIC_ASSERT_INCOMPATIBLE_ENUMS_OP(Two , !=, Eins);
DFL_STATIC_ASSERT_INCOMPATIBLE_ENUMS_OP(Zero , <, Zwei);
DFL_STATIC_ASSERT_INCOMPATIBLE_ENUMS_OP(One , >, Null);
