470 template <class R>
471 inline bool OutcomeProvider<T_ResultData>::doNotificationWithResult(const OutcomeHandlerPtr& handler)
472 {
473  if (m_result)
474  {
475  try
476  {
477  if constexpr(std::is_move_constructible_v<R>)
478  {
479  Outcome<T_ResultData> result(std::move(*m_result));
480  handler->endedWithOutcome(result);
481  }
482  else
483  {
484  Outcome<T_ResultData> result(*m_result);
485  handler->endedWithOutcome(result);
486  }
487  return true;
488  }
489  catch(std::exception& e)
490  {
491  setOutcomeError(Errors::OutcomeNotConstructible{}) << getOutcomeNotConstructibleMsg() << e.what();
492  }
493  catch(...)
494  {
495  setOutcomeError(Errors::OutcomeNotConstructible{}) << getOutcomeNotConstructibleMsg();
496  }
497  }
498  return false; // note that because of returning 'false' the outcome error that was set in the 'catch' will be notified.
499 }
