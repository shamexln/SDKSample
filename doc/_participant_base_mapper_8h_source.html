226 inline std::shared_ptr<T> ParticipantBaseMapper::mapInstanceIdentifierToGsoapType(const DFL::Mdib::InstanceIdentifier& src)
227 {
228  auto instanceIdentifier = std::make_shared<T>();
229 
230  const auto& root = src.root();
231  instanceIdentifier->Root = nullptr;
232 
233  // Only map root if known, otherwise it is allowed to omit the attribute from the resulting XML
234  if (!root.isUnknown())
235  {
236  instanceIdentifier->Root = std::make_shared<std::string>(root.get().get());
237  }
238 
239  const auto& extension = src.extension().get();
240  instanceIdentifier->Extension = nullptr;
241 
242  // Only map extension if it has at least one character, otherwise it is required to be omitted from the resulting XML
243  if (!extension.empty())
244  {
245  instanceIdentifier->Extension = std::make_shared<std::string>(src.extension().get());
246  }
247 
248  instanceIdentifier->IdentifierName = mapLocalizedTextSequenceToGsoap(src.identifierNames());
249 
250  if (const auto& type = src.type())
251  {
252  instanceIdentifier->Type = mapCodedValueToGsoap(*type);
253  }
254 
255  mapExtensionsToGsoap(instanceIdentifier->ext__Extension, src.extensions());
256
