Detailed Description

template<class Hash>
class DFL::Crypto::MessageDigest::Signature::Verification< Hash >

Verifies digital signatures using a public key.

Template Parameters
Hashhash function like Sha256.

For details on digital signatures see https://en.wikipedia.org/wiki/Digital_signature

Verification<Sha256> verification{publicKey};
for (int i = 0; i < 2; ++i)
{
verification.signature(signatureBytes);
verification.init();
for (const auto& b : messageBytes)
{
verification.update(BytesView{&b, 1});
}
verification.final();
EXPECT_TRUE(verification.passed());
}
