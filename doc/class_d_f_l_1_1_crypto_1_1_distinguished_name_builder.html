Detailed Description

Initializes each field of a distinguished name in separation.

const DistinguishedName ctorDn{CountryName{"DE"},
StateOrProvinceName{},
LocalityName{},
OrganizationName{"Draeger"},
OrganizationalUnitName{},
CommonName{"cn"},
UserId{}};
const DistinguishedName builderDn{DistinguishedName::createDn()
.commonName("cn")
.country("DE")
.organization("Draeger")};
ASSERT_EQ(ctorDn, builderDn);
