195 template<class T_Builder>
196 inline void ParticipantMetricMapper::mapAbstractMetricDescriptorFromGsoap(T_Builder& dest, const pm__AbstractMetricDescriptor& src)
197 {
198  ParticipantBaseMapper::mapAbstractDescriptorFromGsoap(dest, src);
199 
200  if (src.Unit)
201  {
202  dest.unit(ParticipantBaseMapper::mapUnitFromGsoap(*src.Unit));
203  }
204  else
205  {
206  dest.unit(DFL::Mdib::Unit::UNSPECIFIED);
207  }
208 
209  dest.metricAvailability(mapMetricAvailabilityFromGsoap(src.MetricAvailability));
210  dest.metricCategory(mapMetricCategoryFromGsoap(src.MetricCategory));
211  dest.bodySites(ParticipantBaseMapper::mapBodySitesFromGsoap(src.BodySite));
212  dest.relations(mapRelationsFromGsoap(src.Relation));
213 
214  if (src.DerivationMethod)
215  {
216  dest.derivationMethod(mapDerivationMethodFromGsoap(src.DerivationMethod));
217  }
218  else
219  {
220  // Biceps:
221  // The default value SHALL be applied, depending on pm:AbstractDescriptor/@MetricCategory.
222  // - If pm:AbstractDescriptor/@MetricCategory is "Set" or "Preset", then the default value of DerivationMethod is "Man"
223  // - If pm:AbstractDescriptor/@MetricCategory is "Clc", "Msrmt", "Rcmm", then the default value of DerivationMethod is "Auto"
224  // - If pm:AbstractDescriptor/@MetricCategory is "Unspec", then no default value is being implied
225 
226  // The implied value of the Biceps standard is not applied because the requirement 3825 states to not assume an
227  // implied value for DerivationMethod.
228  dest.derivationMethod(boost::none);
229  }
230 
231  dest.maxDelayTime(ParticipantBaseMapper::toOptionalMdibDuration(src.MaxDelayTime));
232  dest.maxMeasurementTime(ParticipantBaseMapper::toOptionalMdibDuration(src.MaxMeasurementTime));
233  dest.determinationPeriod(ParticipantBaseMapper::toOptionalMdibDuration(src.DeterminationPeriod));
234  dest.lifeTimePeriod(ParticipantBaseMapper::toOptionalMdibDuration(src.LifeTimePeriod));
235  dest.activationDuration(ParticipantBaseMapper::toOptionalMdibDuration(src.ActivationDuration));
236 }
