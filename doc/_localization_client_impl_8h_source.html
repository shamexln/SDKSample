131 inline LocalizationClient::Notifier LocalizationClientImpl<SupportedLanguagesJob>::fetchSupportedLanguages()
132 {
133  auto outcomeHandler = this->makeOutcomeHandler([this](OutcomeType& outcome)
134  {
135  if (outcome.hasError())
136  {
137  raiseError(outcome.error());
138  }
139  else if (outcome.result()->mdibVersion.isOtherSequenceOf(m_mdibVersion))
140  {
141  auto error = S31::Utils::makeError(Errors::MdibSequenceIdMismatch{}, "MDIB SequenceId mismatch");
142  error.messageStream() << ": previous MDIB version: " << m_mdibVersion << ", new MDIB version: " << outcome.result()->mdibVersion;
143  raiseError(error);
144  }
145  else
146  {
147  std::lock_guard<std::mutex> const guard(m_mutex);
148  m_supportedLanguages = outcome.result()->languages;
149  S31_STREAM_TRACE_LOGGER("Localization", "GetSupportedLanguagesJob successfully completed");
150  }
151 
152  if (m_fetchSupportLanguageData.notifier)
153  {
154  m_fetchSupportLanguageData.notifier->complete();
155  m_fetchSupportLanguageData.notifier = nullptr;
156  }
157  });
158 
159  auto createJob = [](
160  const S31::Sdc::Impl::Endpoint::Url& url,
161  const S31::Utils::OutcomeTypes::HandlerPtr<Sdc::Impl::GetSupportedLanguagesJobResult>& handler)
162  {
163  return std::make_unique<SupportedLanguagesJob>(handler, url);
164  };
165 
166 
167  std::lock_guard<std::mutex> const guard(m_mutex);
168 
169  // is there a fetch running already?
170  if (m_fetchSupportLanguageData.notifier)
171  {
172  return DFL::asNotNull(m_fetchSupportLanguageData.notifier);
173  }
174 
175  m_fetchSupportLanguageData.handler = m_executor.submit<Sdc::Impl::GetSupportedLanguagesJobResult>(
176  outcomeHandler, createJob);
177 
178  m_fetchSupportLanguageData.notifier = std::make_shared<LocalizationImpl::LocalizationClientRequestCompleteImpl>();
179  return DFL::asNotNull(m_fetchSupportLanguageData.notifier);
180 }

325 {
326  std::lock_guard<std::mutex> const guard(m_mutex);
327 
328  {
