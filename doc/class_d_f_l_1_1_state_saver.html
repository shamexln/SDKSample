The given setter will be used for restoring.

s.intVec.push_back(42);

ASSERT_EQ(1U, s.intVec.size());
ASSERT_EQ(42, s.intVec[0]);
s.intVec.clear();
ASSERT_TRUE(s.intVec.empty());
}
ASSERT_EQ(1U, s.intVec.size());
ASSERT_EQ(42, s.intVec[0]);
