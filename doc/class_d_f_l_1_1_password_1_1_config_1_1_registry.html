static void DFL::Password::Config::Registry::secret ( Store::SecretView  s)
static

Injects a secret to be appended to the securitycode before hashing.

Since the allowed securitycodes have a low entropy a randomly generated secret with a length of at least 32 bytes has to be added to the securitycodes to make them more secure. If no secret is provided a default secret is used. But bear in mind that in that case import/exports of securitycodes works between all products using this default secret.

Exceptions
std::runtime_errorIf secret is too short.
// NOLINTNEXTLINE(cppcoreguidelines-avoid-goto)
ASSERT_THROW(Registry::secret(), std::runtime_error);
// NOLINTNEXTLINE(cppcoreguidelines-avoid-goto)
ASSERT_THROW(Registry::secret(secretSize31), std::runtime_error);
// NOLINTNEXTLINE(cppcoreguidelines-avoid-goto)
ASSERT_THROW(Registry::secret(), std::runtime_error);
Registry::secret(secretSize32);
EXPECT_EQ(secretSize32, Registry::secret());
Registry::secret(secretSize33);
EXPECT_EQ(secretSize33, Registry::secret());
Registry::resetSecret();
// NOLINTNEXTLINE(cppcoreguidelines-avoid-goto)
ASSERT_THROW(Registry::secret(), std::runtime_error);
