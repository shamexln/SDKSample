202 inline bool WaitingMdibChangesObserver::predicateMacthesReceivedChangesNoLock(
203  std::function<bool(const NotNull<MdibSharedPtr> &, const NotNull<ConstMdibChangeSetSharedPtr> &)> predicate,
204  boost::optional<DFL::Mdib::MdibChangeType> changeType,
205  bool trackChanges)
206 {
207  for (auto change = m_changes.begin(); change != m_changes.end(); ++change)
208  {
209  if (changeType && changeType != change->changeType)
210  {
211  // Different change type shall not match
212  continue;
213  }
214  if (predicate(change->mdib, change->changes))
215  {
216  if (!trackChanges)
217  {
218  m_changes.erase(change);
219  }
220  return true;
221  }
222  }
223  return false;
