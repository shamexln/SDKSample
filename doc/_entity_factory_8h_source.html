424 template<class T_Entity>
425 inline NotNull<SharedPtr<T_Entity>> EntityFactory::create(const T_Entity& existingEntity,
426  const NotNull<SharedPtr<typename T_Entity::DescriptorType>>& descriptor,
427  const NotNull<SharedPtr<typename T_Entity::StateType>>& state)
428 {
429  if constexpr (std::is_same_v<T_Entity, AlertSignalVariant>)
430  {
431  // This overload is required to get extra information (MDS vs. VMD) type
432  // info from existing entity to create the new entity. Otherwise, the
433  // type information is lost, and it cannot be inserted in the changeset
434  // afterwards correctly.
435  const auto versionNull(DFL::Mdib::VersionCounter(0));
436  return asNotNull(
437  DFL_MDIB_MAKE_SHARED<AlertSignalVariant>(existingEntity.handle(),
438  descriptor,
439  versionNull,
440  state,
441  versionNull,
442  toHandle(existingEntity.parent()),
443  existingEntity.type()));
444  }
445  else
446  {
447  return create(existingEntity.handle(),
448  descriptor,
449  state,
450  existingEntity.parent());
451  }
452 }
