294  static Timestamp aTimestamp()
295  {
296  // We do not use now() since the resolution is too high for Biceps
297  // timestamp round trips. Biceps typically uses an unsigned int with milliseconds
298  // since 1970.
299  // Rounding now() to milliseconds is also not enough, since the gSOAP
300  // s2xsd_dateTime parser uses float for parsing fractional seconds and does
301  // a rounding which sometimes failes the round trip.
302  const Chrono::Seconds timestamp(1537863721); // 2018-09-25T08:22:01
303  return Chrono::SystemClock::TimePoint(timestamp + Chrono::Milliseconds(333));
304  }
308  static Timestamp anotherTimestamp()
309  {
310  const Chrono::Seconds timestamp(1537993721);
311  return Chrono::SystemClock::TimePoint(timestamp + Chrono::Milliseconds(111));
312  }
